Script para criar 10 usuários reais + gerar perfil

# Quantidade de usuarios
$total = 10

for ($i = 1; $i -le $total; $i++) {

    $user = "Userfake$i"
    $senha = "123456"

    Write-Host "Criando usuario: $user" -ForegroundColor Cyan

    # Cria usuario
    net user $user $senha /add /y | Out-Null

    # Garante que nao precisa trocar senha
    net user $user /passwordchg:no | Out-Null

    # Cria o perfil (isso que faltava no seu teste)
    Write-Host "Gerando perfil para: $user" -ForegroundColor Yellow

    try {
        Start-Process -FilePath "cmd.exe" `
            -ArgumentList "/c echo criando perfil..." `
            -Credential (New-Object System.Management.Automation.PSCredential($user,(ConvertTo-SecureString $senha -AsPlainText -Force))) `
            -WindowStyle Hidden `
            -Wait
    } catch {
        Write-Host "Erro ao gerar perfil: $user" -ForegroundColor Red
    }

    Write-Host "OK: $user criado com perfil" -ForegroundColor Green
    Write-Host ""
}
